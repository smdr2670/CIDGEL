122 int Rsimp(int m, int n, double **A, double *b, double *c,
123  double *x, int *basis, int *nonbasis,
124  double **R, double **Q, double *t1, double *t2){
125  int i,j,k,l,q,qv;
126  int max_steps=20;
127  double r,a,at;
128  void GQR(int,int,double**,double**);
129  max_steps=4*n;
130 
131  for(k=0; k<=max_steps;k++){
132  /*
133  ++ Step 0) load new basis matrix and factor it
134  */
135  for(i=0;i<m;i++)for(j=0;j<m;j++)R0(i,j)=AB0(i,j);
136  GQR(m,m,Q,R);
137 
138  /*
139  ++ Step 1) solving system B'*w=c(basis)
140  ++ a) forward solve R'*y=c(basis)
141  */
142  for(i=0;i<m;i++){
143  Y0(i)=0.0;
144  for(j=0;j<i;j++)Y0(i)+=R0(j,i)*Y0(j);
145  if (R0(i,i)!=0.0) Y0(i)=(CB0(i)-Y0(i))/R0(i,i);
146  else {
147  printf("Warning Singular Matrix Found\n");
148  return LP_FAIL;
149  }
150  }
151  /*
152  ++ b) find w=Q*y
153  ++ note: B'*w=(Q*R)'*Q*y= R'*(Q'*Q)*y=R'*y=c(basis)
154  */
155  for(i=0;i<m;i++){
156  W0(i)=0.0;
157  for(j=0;j<m;j++)W0(i)+=Q0(i,j)*Y0(j);
158  }
159 
160  /*
161  ++ Step 2)find entering variable,
162  ++ (use lexicographically first variable with negative reduced cost)
163  */
164  q=n;
165  for(i=0;i<n-m;i++){
166  /* calculate reduced cost */
167  r=CN0(i);
168  for(j=0;j<m;j++) r-=W0(j)*AN0(j,i);
169  if (r<-zero_tol && (q==n || nonbasis0(i)<nonbasis0(q))) q=i;
170  }
171 
172  /*
173  ++ if ratios were all nonnegative current solution is optimal
174  */
175  if (q==n){
176  if (verbose>0) printf("optimal solution found in %d iterations\n",k);
177  return LP_OPT;
178  }
179  /*
180  ++ Step 3)Calculate translation direction for q entering
181  ++ by solving system B*d=-A(:,nonbasis(q));
182  ++ a) let y=-Q'*A(:,nonbasis(q));
183  */
184  for(i=0;i<m;i++){
185  Y0(i)=0.0;
186  for(j=0;j<m;j++) Y0(i)-=Q0(j,i)*AN0(j,q);
187  }
188 
189  /*
190  ++ b) back solve Rd=y (d=R\y)
191  ++ note B*d= Q*R*d=Q*y=Q*-Q'*A(:nonbasis(q))=-A(:,nonbasis(q))
192  */
193  for(i=m-1;i>=0;i--){
194  D0(i)=0.0;
195  for(j=m-1;j>=i+1;j--)D0(i)+=R0(i,j)*D0(j);
196  if (R0(i,i)!=0.0) D0(i)=(Y0(i)-D0(i))/R0(i,i);
197  else {
198  printf("Warning Singular Matrix Found\n");
199  return LP_FAIL;
200  }
201  }
202  /*
203  ++ Step 4 Choose leaving variable
204  ++ (first variable to become negative, by moving in direction D)
205  ++ (if none become negative, then objective function unbounded)
206  */
207  a=0;
208  l=-1;
209  for(i=0;i<m;i++){
210  if (D0(i)<-zero_tol){
211  at=-1*XB0(i)/D0(i);
212  if (l==-1 || at<a){ a=at; l=i;}
213  }
214  }
215  if (l==-1){
216  if (verbose>0){
217  printf("Objective function Unbounded (%d iterations)\n",k);
218  }
219  return LP_UNBD;
220  }
221  /*
222  ++ Step 5) Update solution and basis data
223  */
224  XN0(q)=a;
225  for(j=0;j<m;j++) XB0(j)+=a*D0(j);
226  XB0(l)=0.0; /* enforce strict zeroness of nonbasis variables */
227  qv=nonbasis0(q);
228  nonbasis0(q)=basis0(l);
229  basis0(l)=qv;
230  }
231  if (verbose>=0){
232  printf("Simplex Algorithm did not Terminate in %d iterations\n",k);
233  }
234  return LP_FAIL;
235 }

239 void GQR(int r, int c, double **Q, double **R){
240  int i,j,k;
241  double s,s1,s2;
242  double t1,t2;
243 
244  for(i=0;i<r;i++){
245  for(k=0;k<r;k++)Q0(i,k)=0.0;
246  Q0(i,i)=1.0;
247  }
248 
249  for (i=0;i<c;i++)
250  for (k=i+1;k<r;k++)
251  /* performing givens rotations to zero A[k][i] */
252  if (R0(k,i)!=0){
253  s=sqrt(R0(i,i)*R0(i,i)+R0(k,i)*R0(k,i));
254  s1=R0(i,i)/s;
255  s2=R0(k,i)/s;
256  for(j=0;j<c;j++) {
257  t1=R0(i,j);
258  t2=R0(k,j);
259  R0(i,j)=s1*t1+s2*t2;
260  R0(k,j)=-s2*t1+s1*t2;
261  }
262  /* actually doing givens row rotations on Q */
263  for(j=0;j<r;j++){
264  t1=Q0(j,i);
265  t2=Q0(j,k);
266  Q0(j,i)=s1*t1+s2*t2;
267  Q0(j,k)=-s2*t1+s1*t2;
268  }
269  }
